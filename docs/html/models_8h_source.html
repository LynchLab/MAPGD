47  inline float_t lnL(const float_t &logMM, const float_t &logMm, const float &logmm, const count_t *count){
48  /*posterior = prior x likelihood */
49  E0_=logMM+lnMMP_.lnprob(count);
50  E1_=logMm+lnMmP_.lnprob(count);
51  E2_=logmm+lnmmP_.lnprob(count);
52 
53  if (E0_>E2_) std::swap(E2_, E0_);
54  if (E1_>E2_) std::swap(E2_, E1_);
55 
56  //We make E2 the largest (i.e. least negative) value of the three. E0 and E1 will often times be
57  //extreamly small, and can even be negative infinity. So we are basically just ensuring that we
58  //don't throw away a lot of precission by returning log(exp(E2) ).
59 
60  return log(exp(E0_-E2_)+exp(E1_-E2_)+1.)+E2_;
61  }
